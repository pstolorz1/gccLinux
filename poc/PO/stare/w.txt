#include <iostream>

using namespace std;

class tablica
{
private:
    int x;
    int ilosc;
    int *tab;
public:
    tablica()
    {
        this->tab = NULL;
    }
    tablica(int ilosc)
    {
        tab=new int[ilosc];
    }

    ~tablica()
    {
        delete[] tab;
        tab=0;
    }

     tablica operator =(const tablica &ilosc2)
    {
        this->ilosc = ilosc2.ilosc;
    }

    resize(int ilosc)
    {
        tab=new int(ilosc);
    }

    tablica operator [](int indeks)
    {
        return (tab[indeks]);
    }
};




int main()
{
    int a=2;
    tablica *tab = new tablica[a];
    return 0;
}
